   userid int PRIMARY KEY AUTO_INCREMENT,
   username VARCHAR(50),
   password VARCHAR(255),
   firstname VARCHAR(50),

   salary FLOAT,
   age INTEGER,
   registerday DATE,
   signintime DATETIME
